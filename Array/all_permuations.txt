#include<bits/stdc++.h>
using namespace std;
void permutations(string s,int n,int x,vector<string> &options)
{
    if(x>=n)
    {
        options.push_back(s);
        return;
    }
    permutations(s+"+",n,x+1,options);
    permutations(s+"-",n,x+1,options);
    permutations(s+"*",n,x+1,options);
    permutations(s+"/",n,x+1,options);
}
int main()
{
    int n;
    cin>>n;
    
    int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    int target;
    cin>>target;
    vector<string> options;
    permutations("",n,1,options);
    for(int i=0;i<options.size();i++)
    {
        for(int j=0;j<n-1;j++)
        {
            cout<<a[j]<<" "<<options[i][j]<<" ";
        }
        cout<<a[n-1];
        cout<<endl;
    }
    return 0;
}
